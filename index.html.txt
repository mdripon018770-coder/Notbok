<script>
    let records = JSON.parse(localStorage.getItem('smart_ledger_time_db')) || [];
    let currentEditId = null;

    function openEditor(id = null) {
        if (id) {
            currentEditId = id;
            let item = records.find(r => r.id === id);
            document.getElementById('name').value = item.name;
            document.getElementById('phone').value = item.phone || '';
            document.getElementById('text').value = item.text;
            document.getElementById('delIcon').style.display = 'block';
        } else {
            currentEditId = null;
            document.getElementById('name').value = '';
            document.getElementById('phone').value = '';
            document.getElementById('text').value = '';
            document.getElementById('delIcon').style.display = 'none';
        }
        document.getElementById('modal').style.display = 'flex';
    }

    function closeEditor() { document.getElementById('modal').style.display = 'none'; }

    function saveData() {
        let name = document.getElementById('name').value.trim();
        let phone = document.getElementById('phone').value.trim();
        let text = document.getElementById('text').value;

        if (!name) return alert("Enter Name");

        // Time and Date Calculation
        let now = new Date();
        let dateStr = now.toLocaleDateString('en-GB'); // DD/MM/YYYY
        let timeStr = now.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });
        let fullStamp = dateStr + " | " + timeStr;

        let total = 0;
        let nums = text.match(/\d+/g);
        if (nums) nums.forEach(n => total += parseInt(n));

        if (currentEditId) {
            let idx = records.findIndex(r => r.id === currentEditId);
            // Keeping old time or updating it? Usually, we update the timestamp on edit.
            records[idx] = { id: currentEditId, name, phone, text, total, time: fullStamp };
        } else {
            records.push({ id: Date.now(), name, phone, text, total, time: fullStamp });
        }

        localStorage.setItem('smart_ledger_time_db', JSON.stringify(records));
        closeEditor();
        loadData();
    }

    function deleteItem() {
        if (confirm("Delete this record?")) {
            records = records.filter(r => r.id !== currentEditId);
            localStorage.setItem('smart_ledger_time_db', JSON.stringify(records));
            closeEditor();
            loadData();
        }
    }

    function loadData() {
        let list = document.getElementById('ledgerList');
        let search = document.getElementById('searchInput').value.toLowerCase();
        list.innerHTML = '';

        records.filter(r => r.name.toLowerCase().includes(search)).forEach(r => {
            list.innerHTML += `
                <div class="smart-card" onclick="openEditor(${r.id})">
                    <div class="cust-details">
                        <b>${r.name}</b>
                        <small>${r.phone ? '📞 ' + r.phone : 'No Number'}</small>
                        <span class="time-tag"><i class="far fa-clock"></i> ${r.time}</span>
                    </div>
                    <div class="balance-box">
                        <label>Total Due</label>
                        <div>৳ ${r.total}</div>
                    </div>
                </div>
            `;
        });
    }
    loadData();
</script>
