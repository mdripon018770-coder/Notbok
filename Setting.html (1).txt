// Avatar
if(localStorage.getItem("avatar")){
avatar.src=localStorage.getItem("avatar")
}

upload.onchange=function(){

let reader=new FileReader()

reader.onload=function(){
localStorage.setItem("avatar",reader.result)
avatar.src=reader.result
}

reader.readAsDataURL(this.files[0])

}


// Password
if(!localStorage.getItem("password")){
localStorage.setItem("password","123456")
}

function toggle(id){

let input=document.getElementById(id)

if(input.type=="password"){
input.type="text"
}else{
input.type="password"
}

}

function changePassword(){

let old=oldpass.value
let newp=newpass.value
let conf=confirmpass.value
let saved=localStorage.getItem("password")

if(old!=saved){
alert("Old Password Wrong")
return
}

if(newp.length<6 || newp.length>15){
alert("Password must be 6-15 characters")
return
}

if(newp!=conf){
alert("Password not match")
return
}

localStorage.setItem("password",newp)

alert("Password Changed Successfully")

}


// Profile Save

if(localStorage.getItem("dob")){
dob.value=localStorage.getItem("dob")
}

if(localStorage.getItem("about")){
about.value=localStorage.getItem("about")
}

function saveProfile(){

localStorage.setItem("dob",dob.value)
localStorage.setItem("about",about.value)

alert("Profile Saved")

}


// KYC

function verify(){

if(localStorage.getItem("kyc")=="done"){
alert("Already Verified")
return
}

if(!nidfront.files[0] || !nidback.files[0]){
alert("Upload both NID images")
return
}

localStorage.setItem("kyc","done")

document.getElementById("kycstatus").innerHTML="✅ Verified"

}

if(localStorage.getItem("kyc")=="done"){
document.getElementById("kycstatus").innerHTML="✅ Verified"
}
